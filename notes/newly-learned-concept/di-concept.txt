injection token and configuration token



in app.config


export interface appConfig {
    api_url : string,
    app_version:string;
    featureFlag : boolean;
}



export const APP_CONFIG = new InjectionToken<appConfig>('app.config',{
    provideIn : 'root',
    factory : () => ({
        api_url : 'http',
        app_version : 20,
        featureFlag : true,
    })

})


create a service for seprate appconfig 

private appConfig = inject(APP_CONFIG)


in that service use that app_config
